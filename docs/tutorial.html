Tutorial: Hello Solar System

Build a rotating cube, add lights, a camera, hook it up to audio, and ship it — in 5 steps.

1
Create a new package

Artlab scenes are self-contained packages: a directory with a manifest, DSL source files, and any assets (textures, audio). The CLI scaffolds everything for you.

Terminal
# Install the CLI if you haven't already
npm install -g artlab-cli

# Scaffold a new scene package called "my-scene"
artlab create my-scene

# Start the development server with live reload
artlab serve my-scene

Open http://localhost:4000 — you should see a black canvas with a default environment. The package directory looks like this:

Package layout
my-scene/
  artlab.json      # manifest (name, entry, assets, dependencies)
  src/
    main.art       # DSL entry point
  assets/          # textures, audio, etc.
  dist/            # compiled output (managed by CLI)

Open src/main.art in any editor — changes are picked up automatically while artlab serve is running.

2
Write a simple rotating cube

Replace the contents of src/main.art with the following. Two functions are special: setup() is called once when the scene loads, and update(elapsed, dt) is called every frame.

src/main.art
use "artlab/geometry" as geo

# Declare a module-level variable to hold the cube mesh
let cube: mesh = null

fn setup() {
  # Create a 2x2x2 box and add it to the scene
  cube = geo.add(scene, geo.box(2, 2, 2), {
    color:     0x4488ff,
    roughness: 0.4,
    metalness: 0.1,
  })
}

fn update(elapsed: num, dt: num) {
  # Rotate the cube around the Y axis at 0.5 rad/s
  cube.rotation.y = elapsed * 0.5
}
What you should see: A blue-ish cube slowly rotating against a black background. It appears flat because there are no lights yet.
How it compiles
// DSL: geo.add(scene, geo.box(2,2,2), { color: 0x4488ff, ... })
// JS:  const _geo = await import('/artlab/stdlib/geometry.js')
//       _geo.add(scene, _geo.box(2, 2, 2), { color: 0x4488ff, roughness: 0.4, metalness: 0.1 })
3
Add lights and a camera

Import artlab/lights for scene illumination and artlab/cameras for mouse-controlled orbit navigation.

src/main.art (updated)
use "artlab/geometry" as geo
use "artlab/lights"   as L
use "artlab/cameras"  as cam

let cube: mesh   = null
let camera        = null

fn setup() {
  # Geometry
  cube = geo.add(scene, geo.box(2, 2, 2), {
    color:     0x4488ff,
    roughness: 0.4,
    metalness: 0.1,
  })

  # Lighting
  L.addLight(scene, L.directional(0xffffff, 2), [10, 20, 10])
  L.addLight(scene, L.ambient(0x334466, 0.6))

  # Orbit camera — drag to rotate, scroll to zoom
  camera = cam.OrbitCamera(renderer)
}

fn update(elapsed: num, dt: num) {
  cube.rotation.y = elapsed * 0.5
  camera.update(dt)
}
What you should see: A properly lit blue cube with ambient fill. Left-click drag to orbit, right-click drag to pan, scroll wheel to zoom.
4
Add audio reactivity

Make the cube scale with the bass frequency of the microphone (or any audio source). The artlab/audio module wraps the Web Audio API and Tone.js.

Note: The Web Audio API requires a user gesture before the audio context can start. We attach a click listener in setup().
src/main.art (updated)
use "artlab/geometry"  as geo
use "artlab/lights"    as L
use "artlab/cameras"   as cam
use "artlab/audio"    as audio
use "artlab/animation" as anim

let cube:   mesh = null
let camera        = null

fn setup() {
  # Geometry + material
  cube = geo.add(scene, geo.sphere(1.5, 64), {
    color:     0x22aaff,
    roughness: 0.2,
    metalness: 0.7,
  })

  # Lights
  L.addLight(scene, L.directional(0xffffff, 2.5), [10, 20, 10])
  L.addLight(scene, L.hemisphere(0x88aaff, 0x221133, 0.8))

  # Camera
  camera = cam.OrbitCamera(renderer)

  # Start audio on first click
  canvas.addEventListener("click", fn() { audio.start() })

  # Bind bass band (0..1) to scale of the sphere, mapped to 0.8..2.5
  anim.audioReact(cube.scale, "x", "bass", { min: 0.8, max: 2.5 })
  anim.audioReact(cube.scale, "y", "bass", { min: 0.8, max: 2.5 })
  anim.audioReact(cube.scale, "z", "bass", { min: 0.8, max: 2.5 })
}

fn update(elapsed: num, dt: num) {
  # Slow Y rotation, faster on high frequencies
  let h = audio.band("high")
  cube.rotation.y = elapsed * (0.3 + h * 1.5)

  # Drive animation system and audio analysis
  audio.update()
  anim.animations.update(dt)
  camera.update(dt)
}
What you should see: A metallic sphere that pulses in size with the bass frequencies and spins faster during high-frequency moments. Click the canvas first to unlock the audio context.
5
Pack and share

When you are happy with the scene, bundle it into a portable .artpack.zip. The archive is self-contained — all source, compiled output, and assets are included.

Terminal
# Build and pack the scene
artlab pack my-scene

# Output:
#   Compiling my-scene...   done
#   Bundling assets...      done
#   Writing my-scene.artpack.zip
#   Bundle size: 48 KB (gzip: 18 KB)

The resulting .artpack.zip can be:

  • › Hosted on any static file server (the artlab player fetches and runs it).
  • › Shared as a URL: https://artlab.run/play?src=https://you.example/my-scene.artpack.zip
  • › Embedded in a page with the <artlab-player src="..."> web component.
Embed in HTML
<script type="module" src="https://cdn.artlab.run/player.js"></script>
<artlab-player src="./my-scene.artpack.zip" width="800" height="600"></artlab-player>

That's it. From here, explore the Runtime API to add orbital physics, particle emitters, cinematic cameras, and more.
